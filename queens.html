(define (enum a b)
  (if (> a b)
      '()
      (cons a (enum (+ a 1) b))))
(define (queens n)
  (define choices (enum 1 n))
  (define (valid? sol)
    (let ((c (car sol))
          (r (cdr sol)))
      (let loop ((l1 (+ c 1))
                 (l2 c)
                 (l3 (- c 1))
                 (rest r))
        (cond ((null? rest) #t)
              ((or (= l1 (car rest))
                   (= l2 (car rest))
                   (= l3 (car rest)))
               #f)
              (else (loop (+ l1 1)
                          l2
                          (- l3 1)
                          (cdr rest)))))))
  (define (q k)
    (if (= k 0)
        '(())
        (filter valid?
                (append-map (lambda (sol)
                              (map (lambda (c) (cons c sol)) choices))
                            (q (- k 1))))))
  (q n))
