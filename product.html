product

;(append-map proc lst)等价于(apply append (map proc lst))
;其他语言常以flatmap称append-map
(define (product . lst*)
  (let loop ((rest lst*) (result '(())))
    (if (null? rest)
        result
        (loop (cdr rest)
              (append-map (lambda (d)
                            (map (lambda (a) (cons a d))
                                 (car rest)))
                          result)))))
