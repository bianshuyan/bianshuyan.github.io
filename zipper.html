(define (make-loc focus path)
  (vector 'loc focus path))
(define (loc-focus loc)
  (vector-ref loc 1))
(define (loc-path loc)
  (vector-ref loc 2))
(define (loc? x)
  (and (vector? x)
       (= (vector-length x) 3)
       (eq? (vector-ref x 0) 'loc)))
(define (make-path left up right)
  (vector 'path left up right))
(define (path-left path)
  (vector-ref path 1))
(define (path-up path)
  (vector-ref path 2))
(define (path-right path)
  (vector-ref path 3))
(define (path? x)
  (or (eq? x 'top)
      (and (vector? x)
           (= (vector-length x) 4)
           (eq? (vector-ref x 0) 'path))))
(define (build-loc tree)
  (make-loc tree 'top))
(define (go-left loc)
  (unless (loc? loc)
    (error 'go-left "~s is not a loc" loc))
  (let ((focus (loc-focus loc))
        (path (loc-path loc)))
    (when (eq? path 'top)
      (error 'go-left "left of top"))
    (let ((left (path-left path))
          (up (path-up path))
          (right (path-right path)))
      (if (null? left)
          (error 'go-left "left of first")
          (let ((a (car left))
                (d (cdr left)))
            (make-loc a (make-path d up (cons focus right))))))))
(define (go-right loc)
  (unless (loc? loc)
    (error 'go-right "~s is not a loc" loc))
  (let ((focus (loc-focus loc))
        (path (loc-path loc)))
    (when (eq? path 'top)
      (error 'go-right "right of top"))
    (let ((left (path-left path))
          (up (path-up path))
          (right (path-right path)))
      (if (null? right)
          (error 'go-right "right of last")
          (let ((a (car right))
                (d (cdr right)))
            (make-loc a (make-path (cons focus left) up d)))))))
(define (go-up loc)
  (unless (loc? loc)
    (error 'go-up "~s is not a loc" loc))
  (let ((focus (loc-focus loc))
        (path (loc-path loc)))
    (when (eq? path 'top)
      (error 'go-up "up of top"))
    (let ((left (path-left path))
          (up (path-up path))
          (right (path-right path)))
      (define focus^
        (append (reverse left) (cons focus right)))
      (make-loc focus^ up))))
(define (go-down loc)
  (unless (loc? loc)
    (error 'go-down "~s is not a loc" loc))
  (let ((focus (loc-focus loc))
        (path (loc-path loc)))
    (unless (pair? focus)
      (error 'go-down "down of leaf"))
    (let ((focus^ (car focus))
          (left^ '())
          (right^ (cdr focus)))
      (make-loc focus^ (make-path left^ path right^)))))
(define (nth-down loc n)
  (unless (loc? loc)
    (error 'nth-down "~s is not a loc" loc))
  (let ((focus (loc-focus loc))
        (path (loc-path loc)))
    (unless (pair? focus)
      (error 'nth-down "down of leaf"))
    (let loop ((focus^ (car focus))
               (left^ '())
               (right^ (cdr focus))
               (c n))
      (cond ((= c 0) (make-loc focus^ (make-path left^ path right^)))
            ((null? right^) (error 'nth-down "right of last"))
            (else (loop (car right^)
                        (cons focus^ left^)
                        (cdr right^)
                        (- c 1)))))))
